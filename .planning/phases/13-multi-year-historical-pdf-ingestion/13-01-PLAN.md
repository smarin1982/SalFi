def _make_absolute(href: str, base_origin: str) -> str:
    if href.startswith("http"):
        return href
    if href.startswith("//"):
        scheme = base_origin.split("://")[0]
        return f"{scheme}:{href}"
    if href.startswith("/"):
        return f"{base_origin}{href}"
    return f"{base_origin}/{href}"

def _is_on_domain(url: str, target_domain: str) -> bool:
    try:
        return urlparse(url).netloc.lower().lstrip("www.").endswith(target_domain)
    except Exception:
        return False
```

def _extract_year_from_text(text: str) -> Optional[int]:
    """Extract fiscal year (2015–current) from URL or link text."""
    current = datetime.now().year
    for m in re.findall(r'\b(20[12]\d)\b', text):
        y = int(m)
        if 2015 <= y <= current:
            return y
    return None
```

This is a synchronous public function that wraps `_thread_collect_listing_pdfs` in a `ThreadPoolExecutor(max_workers=1)` with a 180-second timeout. Pattern:
```python
def collect_listing_pdfs(listing_url: str, domain: str) -> dict[int, str]:
    with concurrent.futures.ThreadPoolExecutor(max_workers=1) as executor:
        future = executor.submit(_thread_collect_listing_pdfs, listing_url, domain)
        try:
            return future.result(timeout=180)
        except Exception as e:
            logger.warning(f"collect_listing_pdfs failed for {listing_url}: {e}")
            return {}
```

`_thread_collect_listing_pdfs` sets up `asyncio.ProactorEventLoop` on Windows (MANDATORY — see STATE.md [v2.0 Roadmap] decision):
```python
def _thread_collect_listing_pdfs(listing_url: str, domain: str) -> dict[int, str]:
    loop = asyncio.new_event_loop()
    asyncio.set_event_loop(loop)
    if hasattr(asyncio, "WindowsProactorEventLoopPolicy"):
        asyncio.set_event_loop_policy(asyncio.WindowsProactorEventLoopPolicy())
        loop = asyncio.ProactorEventLoop()
        asyncio.set_event_loop(loop)
    try:
        return loop.run_until_complete(_async_collect_listing_pdfs(listing_url, domain))
    finally:
        loop.close()
```

`_async_collect_listing_pdfs` uses async_playwright:
- Parse `listing_url` to get `base_origin` and `target_domain`
- Use a nested `_harvest_page(url)` coroutine that:
  - Navigates to URL with timeout=25_000, wait_until="domcontentloaded"
  - Tries `page.wait_for_load_state("networkidle", timeout=10_000)`, catches PlaywrightTimeout silently
  - Collects all `a[href]` elements
  - For each link: get href, skip non-PDF, skip `_is_partial_year_url()`, extract year, get `_score_pdf_link()`, store `(score, abs_url)` in `candidates: dict[int, list[tuple[int, str]]]`
- After initial harvest, if `len(candidates) < 3`, look for pagination links matching `["page=", "pagina=", "p=", "/page/", "/pagina/"]` in href OR `["anterior", "siguiente", "older", "newer"]` in link text; follow up to 3 pagination pages
- Return `{year: max(candidates, key=lambda x: x[0])[1] for year, candidates in candidates.items() if candidates}`

**Section 6 — LatamBackfiller class:**

```python
class LatamBackfiller:
    def __init__(self, slug: str, country: str, storage_path: Path, domain: str):
        self.slug = slug
        self.country = country
        self.storage_path = storage_path
        self.domain = domain
        self.parquet_path = storage_path / "financials.parquet"
        self.raw_dir = storage_path / "raw"

    def get_target_years(self) -> list[int]:
        """Return list of fiscal years to backfill (last 5 completed years)."""
        current = datetime.now().year
        return [current - i for i in range(1, BACKFILL_YEARS + 1)]

    def get_missing_years(self) -> list[int]:
        """Return target years not yet in parquet."""
        existing = _years_already_in_parquet(self.parquet_path)
        return [y for y in self.get_target_years() if y not in existing]

    def run_year(
        self,
        year: int,
        pdf_url: str,
        currency_code: str,
        force_reextract: bool = False,
    ) -> BackfillResult:
        """
        Download + extract one fiscal year. Does NOT write to parquet.
        Returns BackfillResult with extraction_result populated on success.
        Caller (app.py) decides whether to write (auto for Alta/Media, prompt for Baja).
        """
        # Skip check
        if not force_reextract:
            existing = _years_already_in_parquet(self.parquet_path)
            if year in existing:
                return BackfillResult(year=year, status="skipped")

        # Download
        try:
            self.raw_dir.mkdir(parents=True, exist_ok=True)
            scrape_result = _download_pdf(pdf_url, self.raw_dir, self.slug)
            if not scrape_result.ok or not scrape_result.path:
                return BackfillResult(year=year, status="not_found",
                                      error_msg=scrape_result.error)
            pdf_path = scrape_result.path
        except Exception as e:
            return BackfillResult(year=year, status="error", error_msg=str(e))

        # Extract
        try:
            results = _extract(str(pdf_path), currency_code=currency_code,
                               fiscal_year=year, country=self.country)
            if not results:
                return BackfillResult(year=year, status="error",
                                      pdf_path=pdf_path,
                                      error_msg="extract() returned empty list")
            # _extract returns list[ExtractionResult]; take the result matching year
            target = next(
                (r for r in results if r.fiscal_year == year),
                results[0]
            )
            status = "low_conf" if target.confidence == "Baja" else "ok"
            return BackfillResult(
                year=year, status=status, pdf_path=pdf_path,
                extraction_result=target, confidence=target.confidence
            )
        except Exception as e:
            logger.error(f"LatamBackfiller.run_year extract failed year={year}: {e}")
            return BackfillResult(year=year, status="error", pdf_path=pdf_path,
                                  error_msg=str(e))

    def write_year(self, result: BackfillResult) -> bool:
        """
        Write a BackfillResult to parquet via latam_processor.process().
        Returns True on success. Call only after analyst confirms (if low_conf).
        """
        if not result.extraction_result:
            return False
        try:
            _process(
                company_slug=self.slug,
                extraction_result=result.extraction_result,
                country=self.country,
                data_dir=str(self.storage_path.parent.parent),
            )
            return True
        except Exception as e:
            logger.error(f"LatamBackfiller.write_year failed year={result.year}: {e}")
            return False
```
