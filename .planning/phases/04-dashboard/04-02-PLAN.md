```python
# ── Chart builders ────────────────────────────────────────────────────────────

def build_trend_figure(
    df: pd.DataFrame,
    kpi: str,
    year_range: tuple[int, int],
    ticker: str,
) -> go.Figure:
    """Single-company Plotly trend chart. Bloomberg-minimal chrome."""
    meta = KPI_META[kpi]
    filtered = df[
        (df["fiscal_year"] >= year_range[0]) & (df["fiscal_year"] <= year_range[1])
    ].copy()
    color = COMPANY_COLORS.get(ticker, "#4a4a4a")
    fig = go.Figure()
    fig.add_trace(go.Scatter(
        x=filtered["fiscal_year"],
        y=filtered[kpi],
        mode="lines+markers",
        line=dict(color=color, width=2.5),
        marker=dict(size=5),
        hovertemplate=f"%{{x}}: %{{y:{meta['tick_format']}}}<extra></extra>",
    ))
    fig.update_layout(
        template="plotly_white",
        margin=dict(l=0, r=0, t=10, b=0),
        showlegend=False,
        hovermode="x unified",
        height=200,
    )
    fig.update_xaxes(showgrid=False, tickformat="d", dtick=2)
    fig.update_yaxes(showgrid=True, gridcolor="#f0f0f0", zeroline=False, tickformat=meta["tick_format"])
    return fig


def build_comparativo_figure(
    df_hd: pd.DataFrame,
    df_pg: pd.DataFrame,
    kpi: str,
    year_range: tuple[int, int],
) -> go.Figure:
    """Two-company overlay — HD and PG as separate traces on same figure (DASH-01 Comparativo mode)."""
    meta = KPI_META[kpi]
    fig = go.Figure()
    for ticker, df in [("HD", df_hd), ("PG", df_pg)]:
        d = df[
            (df["fiscal_year"] >= year_range[0]) & (df["fiscal_year"] <= year_range[1])
        ]
        fig.add_trace(go.Scatter(
            x=d["fiscal_year"],
            y=d[kpi],
            name=ticker,
            mode="lines+markers",
            line=dict(color=COMPANY_COLORS[ticker], width=2.5),
            marker=dict(size=5),
            hovertemplate=f"{ticker} %{{x}}: %{{y:{meta['tick_format']}}}<extra></extra>",
        ))
    fig.update_layout(
        template="plotly_white",
        title=dict(text=meta["label"], font=dict(size=14, color="#333")),
        showlegend=True,
        legend=dict(orientation="h", yanchor="bottom", y=1.02, x=0),
        hovermode="x unified",
        margin=dict(l=0, r=0, t=50, b=0),
        height=260,
    )
    fig.update_xaxes(showgrid=False, tickformat="d", dtick=2)
    fig.update_yaxes(showgrid=True, gridcolor="#f0f0f0", zeroline=False, tickformat=meta["tick_format"])
    return fig
```

**Helper: render one Executive Card (single-company)**
```python
def render_kpi_card(
    kpi: str,
    df: pd.DataFrame,
    year_range: tuple[int, int],
    ticker: str,
) -> None:
    """
    Executive Card = headline label + big number (latest) + delta pill (YoY %) + Plotly trend chart.
    Uses st.metric for headline/number/delta, then st.plotly_chart for trend.
    """
    meta = KPI_META[kpi]
    col_data = df[kpi].dropna() if kpi in df.columns else pd.Series(dtype=float)

    # Filter to year_range for display
    df_filtered = df[
        (df["fiscal_year"] >= year_range[0]) & (df["fiscal_year"] <= year_range[1])
    ]
    col_filtered = df_filtered[kpi].dropna() if kpi in df_filtered.columns else pd.Series(dtype=float)

    latest_val = col_filtered.iloc[-1] if not col_filtered.empty else None
    prior_val = col_filtered.iloc[-2] if len(col_filtered) >= 2 else None

    if latest_val is not None and prior_val is not None and prior_val != 0:
        delta_pct = (latest_val - prior_val) / abs(prior_val)
    else:
        delta_pct = None

    # Big Number and Delta Pill via st.metric
    st.metric(
        label=f"**{meta['label']}**",
        value=format_kpi(latest_val, meta["format"]),
        delta=format_delta(delta_pct),
        delta_color="normal",  # green=positive, red=negative
        border=True,
    )

    # Historical Trend via Plotly (NOT st.metric sparkline — use Plotly for Bloomberg aesthetic)
    if not col_data.empty:
        fig = build_trend_figure(df, kpi, year_range, ticker)
        st.plotly_chart(fig, width="stretch", config={"displayModeBar": False})
    else:
        st.caption("Sin datos disponibles.")
```

**Critical implementation notes:**

1. `st.plotly_chart(fig, width="stretch")` — NOT `use_container_width=True` (deprecated in Streamlit 1.40+, removed after 2025-12-31).

2. `config={"displayModeBar": False}` — removes Plotly's toolbar chrome for a cleaner Bloomberg aesthetic. No zoom/pan controls cluttering the card.

3. Dynamic layout for 5 KPIs: TWO separate `st.columns` calls (not one call with 5). First call creates a 2-column row; second call creates a 3-column row. Streamlit stacks them vertically, producing the 2+3 layout as designed.

4. For n=1: Do NOT wrap in `st.columns([1])` — render directly for true full-width display.

5. `render_kpi_card` filters data to `year_range` for the big number (latest within range) but passes the full `df` to `build_trend_figure` which does its own filtering — so the trend chart correctly clips to the selected years.

6. Never cache `go.Figure` objects. Always create them inside the render function — Plotly figures are mutable; caching causes trace accumulation across reruns.

7. `st.metric(border=True)` — adds a visual card border. Available in Streamlit 1.44+.

8. ROE edge case: HD ROE can be >100% (e.g. 222.9%). `format_kpi(2.229, 'percentage')` returns `"222.9%"` — correct. Do NOT add capping logic.

9. `agent.FinancialAgent` is imported inside the button handler (lazy import) to avoid slowing down the initial page load. The import of `agent` module triggers no ETL on its own.
