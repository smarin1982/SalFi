```python
@st.cache_data(ttl=300)
def load_latam_kpis(country: str, slug: str):
    """Load LATAM kpis.parquet for a processed company. Returns DataFrame or None."""
    path = pathlib.Path("data") / "latam" / country / slug / "kpis.parquet"
    if not path.exists():
        return None
    try:
        import pandas as pd
        return pd.read_parquet(path)
    except Exception:
        return None
```
